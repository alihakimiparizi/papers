  \headerbox{Word Usage Similarity (USim)
}{name=usim,column=0,row=0,span=2}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\Large
\begin{itemize}
    \item Measure the similarity of a pair of usages of a target lemma
      \begin{itemize}
    \item Does not rely on a sense inventory
    \item Well-suited to text types with many OOVs and new senses, e.g.,
      social media
      \end{itemize}
\item \textbf{Scale:} (1) completely different -- (5) the same 
\end{itemize}
\vfill
\begin{center}
\begin{tabular}{cl}
Similarity & Usage pair\\
\hline
\multirow{2}{*}{3.3} 
& \large{A chain is only as \underline{strong} as its weakest link.}\\
&  \large{Norway must send a \underline{strong} signal.} \\
 &  \\
\multirow{2}{*}{1.3} 
&  \large{A country needs a \underline{strong} leader.}\\
&  \large{It has a \underline{strong} fruity flavour.}\\
\end{tabular}
\end{center}
\vfill

}

  \headerbox{Embeddings
}{name=embeddings,column=0,row=0,span=2, below=usim}{
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\Large    

\begin{itemize}
\item Considered 4 embedding-based representations of a single target usage

\begin{itemize}
%%     \item \textbf{Embeddings:} Vector representations of text generated using neural networks 
%% 	\vfill    
%% \setlength{\itemindent}{+.2in}
\item \textbf{Word2vec:} Average skipgram embeddings for words in
  target sentence
\item \textbf{GloVe:} Average GloVe embeddings for words in
target sentence
\item \textbf{Skip-thoughts:} Embed target sentence 
\item \textbf{Context2vec:} Embed context of target word in sentence
\vfill
\end{itemize}
\end{itemize}
